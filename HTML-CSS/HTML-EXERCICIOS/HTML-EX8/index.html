class Pessoa:
    def __init__(self, nome, idade):
        self.nome = nome
        self.idade = idade
    def mostrar(self):
        print(f"{self.nome} tem {self.idade} anos.")

p1 = Pessoa("Luiz", 20)
p1.mostrar()
